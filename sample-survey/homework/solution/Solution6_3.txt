x <- c(110, 265, 360, 1265)

psi <- x / sum(x)

y <- c(44, 82, 112, 362)

t <- sum(y);  t;  # 600

# E(t.hat.psi) = t = 600

sum(psi * (y/psi - t)^2) / n

# V(t.hat.psi) = 1410.12


# Part (b)

# E(t.hat.srs) = t = 600

N^2 * var(y)/n * (1 - n/N)
